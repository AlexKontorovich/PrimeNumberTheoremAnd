\begin{abstract}
This note collects discrepancies found while matching statements on the
\href{https://archimede.pages.math.cnrs.fr/tme-emt-wiki/index.html}{TME--EMT wiki}
against the cited source papers, in the course of formalizing those
statements in Lean.  Items are grouped by wiki page.  Only high-confidence
mismatches are listed; each entry gives the wiki text, the source, and a
proposed correction.
\end{abstract}

\section*{How to read this note}
Citations of the form ``Art01'' refer to
\href{https://archimede.pages.math.cnrs.fr/tme-emt-wiki/Art01.html}{Art01.html}
(``Explicit bounds on primes'').  Page numbers for papers are from the
versions available at the URLs recorded in the wiki bibliography.

\section{Art01 --- Explicit bounds on primes}

\subsection{Del\'eglise--Nicolas 2019, $\pi_2$ upper error constant}
\label{err:dn-11181}
\paragraph{Wiki.}
Art01 quotes, for $x\ge 60173$,
\[
\pi_2(x)-\Bigl(\tfrac{x^3}{3\log x}+\tfrac{x^3}{9\log^2 x}+\tfrac{2x^3}{27\log^3 x}\Bigr)
\le \tfrac{11181\,x^3}{648\log^4 x}.
\]
\paragraph{Source.}
Del\'eglise--Nicolas, \emph{The Landau function and the Riemann hypothesis},
arXiv:1907.07664, Corollary~2.7, display~(2.21), uses the constant
\textbf{1181} (not 11181).  The same constant appears in the simplification
leading to~(2.22), as the coefficient $1181/216$.
\paragraph{Proposed correction.}
Replace $11181$ by $1181$.

\subsection{Platt--Trudgian 2021, $\psi$ versus $\pi$}
\label{err:pt2021-psi}
\paragraph{Wiki.}
Art01 attributes to Platt--Trudgian~(2021) the bound
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 235\,(\log x)^{0.52}\exp\Bigl(-\sqrt{\tfrac{\log x}{5.573412}}\Bigr)
\qquad(x\ge e^{2000}).
\]
\paragraph{Source.}
Platt--Trudgian, \emph{The Riemann hypothesis is true up to}~$3\cdot 10^{12}$,
Table~1 at $X=2000$ and Theorem~1 give a $\psi$-bound of the shape
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 411.4\,\Bigl(\tfrac{\log x}{R}\Bigr)^{1.52}
\exp\bigl(-1.89\sqrt{\log x/R}\bigr),\qquad R=5.573412.
\]
The constants $235$ and exponent $0.52$ belong to the companion $\pi$-bound
(Corollary~2), not to the $\psi$-bound.
\paragraph{Proposed correction.}
Quote the $\psi$-bound from Table~1 / Theorem~1, or explicitly relabel the
displayed estimate as a $\pi$-bound and cite Corollary~2.

\subsection{Johnston--Yang 2023, exponent on $\log x$}
\label{err:jy-1515}
\paragraph{Wiki.}
Art01 quotes
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 9.39\,(\log x)^{1.51}\exp\bigl(-0.8274\sqrt{\log x}\bigr)
\qquad(x\ge 2).
\]
\paragraph{Source.}
Johnston--Yang, Theorem~1.1, uses the exponent $\mathbf{1.515}$.
\paragraph{Proposed correction.}
Replace $1.51$ by $1.515$.

\subsection{De~Koninck--Letendre 2020, missing $k$ in $\log\log k$}
\label{err:dkl-loglog}
\paragraph{Wiki.}
Art01, \S4, writes
\[
\sum_{i\le k}\log\log p_i
\ge k\Biggl(\log\log k+\frac{\log\log - 3/2}{\log k}\Biggr)
\qquad(k\ge 6),
\]
with a bare $\log\log$ in the numerator.
\paragraph{Source.}
De~Koninck--Letendre, arXiv:1812.09950, Lemma~4.8~(4.21):
\[
\sum_{i\le k}\log\log p_i
\ge k\Biggl(\log\log k+\frac{\log\log k - 3/2}{\log k}\Biggr)
\qquad(k\ge 6).
\]
\paragraph{Proposed correction.}
Replace the bare $\log\log$ by $\log\log k$.

\subsection{B\"uthe 2016, omitted $\pi^*$ bound}
\label{err:buthe-pistar}
\paragraph{Wiki.}
Art01, \S1, under B\"uthe~(2016), lists RH-conditional bounds for
$\psi$, $\vartheta$, and $\pi$ only.
\paragraph{Source.}
B\"uthe also states
\[
\bigl|\pi^*(x)-\mathrm{li}(x)\bigr|
\le \tfrac{\sqrt{x}}{8\pi}\log x
\qquad(x>59),
\]
under the same height hypothesis.
\paragraph{Proposed correction.}
Add the $\pi^*$ bound to the displayed list.

\subsection{Axler 2019, Mandl $B_n$ mislabelled as $\sum p_i$}
\label{err:axler-mandl}
\paragraph{Wiki.}
Art01, \S4, attributes to Axler~(2019) the bounds
\[
\sum_{i\le k}p_i
\;\gtrless\;
\frac{k^2}{4}+\frac{k^2}{4\log k}
-\frac{k^2(\log\log k-c)}{4(\log k)^2}
\]
(with $c=2.9$ for the lower bound at $k\ge 6309751$, and $c=4.42$ for the
upper bound at $k\ge 256376$).
\paragraph{Source.}
Axler, \emph{On the sum of the first $n$ prime numbers},
arXiv:1606.06874, Theorems~1.6 and~1.7 bound Mandl's quantity
\[
B_n=\frac{n\,p_n}{2}-\sum_{k\le n}p_k,
\]
not $\sum p_k$.  The actual prime-sum estimates are Theorems~1.8 and~1.9
(leading factor $n^2/(2\log n)$, different constants and ranges).
\paragraph{Proposed correction.}
Replace $\sum_{i\le k}p_i$ by $B_k$ (and cite Theorems~1.6--1.7), or else
quote Theorems~1.8--1.9 for the prime-sum bounds.

\subsection{Cosmetic / typesetting (lower priority)}

\item Art01, Cipolla expansion: mixes $\ln$ and $\log$.
\item Art01, Kadiri~(2008) table (\S6): the entry under

\end{itemize}

\section{Art09 --- Short intervals containing primes}

\subsection{Carneiro--Milinovich--Soundararajan 2019, interval orientation}
\label{err:cms-orientation}
\paragraph{Wiki.}
Art09 quotes, under RH and for $x\ge4$, a prime in
\[
\bigl]x-\tfrac{22}{25}\sqrt{x}\log x,\,x\bigr].
\]
\paragraph{Source.}
Carneiro--Milinovich--Soundararajan, \emph{Fourier optimization and prime
gaps}, Comment.\ Math.\ Helv.\ 94 (2019); arXiv:1708.04122, Theorem~5:
\[
\bigl[x,\,x+\tfrac{22}{25}\sqrt{x}\log x\bigr].
\]
(The backward interval of that shape is Dudek's earlier result with
constant $4/\pi$; Carneiro et al.\ improve the constant and reverse the
orientation.)
\paragraph{Proposed correction.}
Replace the displayed interval by
$[x,\,x+\tfrac{22}{25}\sqrt{x}\log x]$.

\subsection{Trudgian 2016, Corollary~2 threshold}
\label{err:trudgian-cor2}
\paragraph{Wiki.}
Art09 quotes Trudgian~(2016) for $x>2{,}898{,}242$.
\paragraph{Source.}
Trudgian, \emph{Updating the error term in the prime number theorem},
Ramanujan J.\ 39 (2016); arXiv:1401.2689, Corollary~2: $x\ge 2{,}898{,}239$.
\paragraph{Proposed correction.}
Replace $2{,}898{,}242$ by $2{,}898{,}239$, and use the non-strict
threshold $x\ge\cdots$ as in the paper.

\subsection{Dudek 2014/2016, short-interval threshold}
\label{err:dudek-threshold}
\paragraph{Wiki.}
Art09 quotes Dudek for a prime in $(x,x+3x^{2/3}]$ when
$x\ge\exp(\exp(34.32))$.
\paragraph{Source.}
Dudek, \emph{An explicit result for primes between cubes},
Funct.\ Approx.\ Comment.\ Math.\ 55 (2016); arXiv:1401.4233, proves the
short-interval claim under
\[
x^{1/3}>\exp(\exp(33.217))
\]
(equivalently $x>\exp(3\exp(33.217))$), as the $x$-form of the main
cubes theorem $n\ge\exp(\exp(33.217))$.  The wiki value $34.32$ is a
rounded stand-in for $\ln 3+33.217\approx 34.3156$ and is slightly too
large.
\paragraph{Proposed correction.}
Replace the hypothesis by $x^{1/3}>\exp(\exp(33.217))$ (or the exact
equivalent $x>\exp(3\exp(33.217))$).

\subsection{Cully--Hugill 2021, short-interval threshold}
\label{err:cully-hugill-threshold}
\paragraph{Wiki.}
Art09 quotes Cully--Hugill for $(x,x+3x^{2/3}]$ when
$x\ge\exp(\exp(33.99))$.
\paragraph{Source.}
Cully--Hugill, \emph{Primes between consecutive powers},
arXiv:2107.14468, states $x\ge\exp(\exp(33.990))$ for that interval
(with $m=3$).
\paragraph{Proposed correction.}
Replace $33.99$ by $33.990$.

\section{Art01 --- Ramaré 2013 (additional)}

\subsection{Ramaré, original corollary versus 2023 corrigendum}
\label{err:ramare-thresholds}
\paragraph{Wiki.}
Art01, after noting that ``there was an error in this paper which is
corrected below'', quotes for $x\ge 1\,520\,000$ the error
$O^*(0.0067/\log x)$, and for $x\ge 468\,000$ the error
$O^*(0.01/\log x)$.
\paragraph{Source.}
The original Ramaré corollary (Acta Arith.\ 159 (2013)) stated
$O^*(0.0067/\log x)$ for $x\ge 23$, with no $0.01/\log x$ line.
The 2023 corrigendum (sign errors and a computer-program typo for
checks with $x\le 10\,000$) restates the corollary with
$x\ge 1.52\cdot 10^6$ for $0.0067/\log x$, $x\ge 468\,000$ for
$0.01/\log x$, and an additional line $O^*(1/(4\log x))$ for
$x\ge 115$.  The wiki matches the corrigendum (but omits the
$x\ge 115$ line).
\paragraph{Proposed correction.}
Cite the 2023 corrigendum next to the revised thresholds, mention that
the 2013 printed corollary had $x\ge 23$ for $0.0067/\log x$, and add
the $x\ge 115$ bound.

\section{Art06 --- Bounds on $|\zeta(s)|$}

\subsection{Patel 2022, first term $(3/4)\log t$ versus $\log t$}
\label{err:patel-log}
\paragraph{Wiki.}
Art06 quotes
\[
|\zeta(1+it)|\le\min\Bigl(\tfrac34\log t,\;
\tfrac12\log t+1.93,\;\tfrac15\log t+44.02\Bigr)
\qquad(t\ge 3).
\]
\paragraph{Source.}
Patel, arXiv:2009.00769, Theorem~1.1:
\[
|\zeta(1+it)|\le\min\bigl(\log t,\;
\tfrac12\log t+1.93,\;\tfrac15\log t+44.02\bigr).
\]
The factor $\tfrac34$ belongs to Trudgian's earlier bound, also listed
on Art06.
\paragraph{Proposed correction.}
Replace $\tfrac34\log t$ by $\log t$ in the Patel display.

\subsection{Hiary 2016 constant $0.63$}
\label{err:hiary-063}
\paragraph{Wiki.}
Art06 quotes $|\zeta(1/2+it)|\le 0.63\,t^{1/6}\log t$ for $t\ge 3$,
citing Hiary~(2016).
\paragraph{Source.}
That constant used an incorrect form of the Kusmin--Landau lemma.
Hiary--Patel--Yang, arXiv:2207.02366, note that the corrected Hiary
constant is $0.77$, and prove the improved bound with constant
$\mathbf{0.618}$.
\paragraph{Proposed correction.}
Replace $0.63$ by $0.618$ and cite Hiary--Patel--Yang, or else state
explicitly the corrected Hiary constant $0.77$.

\subsection{Rosser 1941, threshold $T\ge 2$ versus $T\ge 1467$}
\label{err:rosser-threshold}
\paragraph{Wiki.}
Art06 quotes Rosser's $N(T)$ bound for $T\ge 2$.
\paragraph{Source.}
Rosser (1941), as recorded e.g.\ in Trudgian's tables, proves
$|S(T)|\le 0.137\log T+0.443\log\log T+1.588$ for
$\mathbf{T\ge 1467}$.
\paragraph{Proposed correction.}
Replace $T\ge 2$ by $T\ge 1467$.

\subsection{Trudgian 2014, extraneous $1/(5T)$}
\label{err:trudgian-N-5T}
\paragraph{Wiki.}
Art06 appends $+1/(5T)$ to the $N(T)$ error for Trudgian~(2014).
\paragraph{Source.}
The published result (J.\ Number Theory 134) is
$|S(T)|\le 0.112\log T+0.278\log\log T+2.510$ for $T\ge e$, with no
$1/(5T)$ term.  (An arXiv corollary has a constant $0.2/T_0$ when
$T\ge T_0\ge e$, which is not $1/(5T)$.)
\paragraph{Proposed correction.}
Drop the $1/(5T)$ summand (or replace it by the parameterized
$0.2/T_0$ form, citing the corollary carefully).

\subsection{Lehman 1970, $t\ge 1/5$ versus $64/(2\pi)$}
\label{err:lehman-threshold}
\paragraph{Wiki.}
Art06 states Lehman's bound for $t\ge 1/5$, noting that Lehman wrote
$t\ge 64/(2\pi)$ and that $1/5$ is a modern computational extension.
\paragraph{Source.}
Lehman, Lemma~2: $t\ge 64/(2\pi)$.
\paragraph{Proposed correction.}
Quote $t\ge 64/(2\pi)$ for the paper statement; keep $t\ge 1/5$ only if
explicitly labelled as a verified extension.

\subsection{Delange 1987, complex versus real}
\label{err:delange-real}
\paragraph{Wiki.}
Art06 quotes, for $\sigma>1$ and any real $t$,
\[
-\Re\frac{\zeta'}{\zeta}(\sigma+it)
\le \frac1{\sigma-1}-\frac1{2\sigma^2}.
\]
\paragraph{Source.}
Delange, Colloq.\ Math.\ 53 (1987), proves for \textbf{real} $s>1$
\[
\frac{\zeta'}{\zeta}(s)+\frac1{s-1}>\frac1{2s^2}
\]
(equivalently the displayed upper bound on $-\zeta'/\zeta(s)$ on the
real line).  The complex form is not in that note.
\paragraph{Proposed correction.}
Restrict the statement to real $s>1$, or cite a different source for
the complex claim.

\subsection{Hasanalizade--Shen--Wong $N(T)$ forms}
\label{err:hsw-forms}
\paragraph{Wiki / transcription risk.}
Art06 lists HSW parameters $0.1038$, $0.2573$, $9.3675$ in the shape that
includes the classical $7/8$ term.
\paragraph{Source.}
Hasanalizade--Shen--Wong, arXiv:2107.06506:
\begin{itemize}[leftmargin=*]
\item Corollary~1.2 (no $7/8$): constant term $9.3675$, for $T\ge e$;
\item Corollary~1.4 (bound on $S(T)=N(T)-\mathrm{main}-7/8$): constant
  terms $8.3675$ and $3.0305$.
\end{itemize}
Pairing $9.3675$ with the $+7/8$ form conflates the two corollaries.
\paragraph{Proposed correction.}
Display Cor.~1.2 and Cor.~1.4 separately, with the matching main terms.

\section{Art08 --- Zero-free regions and related results}

\subsection{Classical zero-free region written with $\log(|\Im s|+2)$}
\label{err:art08-log-plus-two}
\paragraph{Wiki.}
Art08 quotes Kadiri~(2005), Mossinghoff--Trudgian~(2015), and the later
refinement with $R_0=5.558691$, all in the shape
\[
\Re s\ge 1-\frac1{R_0\log(|\Im s|+2)}.
\]
\paragraph{Source.}
The cited papers use $\log|\Im s|$ (equivalently $\log|t|$), not
$\log(|t|+2)$:
\begin{itemize}[leftmargin=*]
\item Kadiri, \emph{Une r\'egion explicite sans z\'ero pour la fonction
  Zeta de Riemann}, arXiv:math/0401238: main theorem
  $\Re s\ge 1-1/(5.70176\log|\Im s|)$ for $|\Im s|\ge 2$.
\item Mossinghoff--Trudgian, arXiv:1410.3926, Theorem~1:
  $\sigma\ge 1-1/(5.573412\log|t|)$ for $|t|\ge 2$.
\item Mossinghoff--Trudgian--Yang, arXiv:2212.06867:
  $\sigma\ge 1-1/(5.558691\log|t|)$ for $|t|\ge 2$.
\end{itemize}
(The same $\log|t|$ form is the one used for the classical zero-free
region predicate in the Lean formalization, following Fiori--Kadiri--Swaminathan.)
\paragraph{Proposed correction.}
Replace $\log(|\Im s|+2)$ by $\log|\Im s|$ in each of these displays
(keeping the stated ranges $|t|\ge 2$).

\subsection{Kadiri $R_0$: $5.70175$ versus $5.70176$}
\label{err:art08-kadiri-r0}
\paragraph{Wiki.}
Art08 attributes to Kadiri~(2002/2005) the constant $R_0=5.70175$.
\paragraph{Source.}
Kadiri, arXiv:math/0401238, states the main theorem with
$R_0=\mathbf{5.70176}$.  (The digit $5.70175$ appears elsewhere in
numerical tables in that paper, but is not the constant of the theorem
statement quoted on Art08.)
\paragraph{Proposed correction.}
Replace $5.70175$ by $5.70176$.

\subsection{Gourdon--Demichel 2004 stated unconditionally}
\label{err:gourdon-conditional}
\paragraph{Wiki.}
Art09 lists ``Theorem (2004, conditional)'' for a prime in
$(x(1-1/14500755538),x]$ when $x>\exp(60)$, but the Lean transcription
previously omitted the RH hypothesis.
\paragraph{Source.}
The Art09 heading itself marks the result as conditional on the
Gourdon--Demichel verification of RH up to height
$T_0\approx 2.44\cdot 10^{12}$.
\paragraph{Proposed correction.}
Record the RH-up-to-$T_0$ hypothesis explicitly alongside the short-interval
claim.

\subsection{Schoenfeld 1976 open versus closed right endpoint}
\label{err:schoenfeld-open}
\paragraph{Wiki / Lean.}
Art09 and the previous Lean statement used a closed interval
$(x,\,x(1+1/16597)]$.
\paragraph{Source.}
Schoenfeld, \emph{Math.\ Comp.}\ 30 (1976), Theorem~12, asserts a prime in
the \textbf{open} interval $(x,\,x+x/16597)$ for $x>2{,}010{,}759.9$.
\paragraph{Proposed correction.}
Use a strict upper bound $p<x+x/16597$.

\subsection{Kadiri--Lumley open right endpoint}
\label{err:kadiri-lumley-open}
\paragraph{Wiki / Lean.}
The previous Lean statement used \texttt{HasPrimeInInterval}, giving
$(1-\Delta^{-1})x < p \le x$.
\paragraph{Source.}
Kadiri--Lumley, arXiv:1407.7902, Theorem~1.1, has
$(1-\Delta^{-1})x < p < x$.
\paragraph{Proposed correction.}
Use a strict upper bound $p<x$.

\subsection{Cheng--Graham large-$t$ constant and Kusmin--Landau}
\label{err:cheng-graham-kusmin}
\paragraph{Wiki / Lean.}
Art06 records $|\zeta(1/2+it)|\le 3\,t^{1/6}\log t$ for $t\ge e$ after
Cheng--Graham~(2004).
\paragraph{Source.}
The same Kusmin--Landau $1/\pi$ versus $2/\pi$ error that invalidated
Hiary's constant $0.63$ (Hiary--Patel--Yang, arXiv:2207.02366) affects the
Cheng--Graham lemmas underlying this large-$t$ bound.
\paragraph{Proposed correction.}
Flag the constant $3$ as provisional (as already done for Hiary on Art06)
until a corrected derivation is available.

\subsection{\texorpdfstring{HSW 2022 threshold $T\ge e$ versus $T\ge 2$}{HSW 2022 threshold T>=e versus T>=2}}
\label{err:hsw-threshold}
\paragraph{Lean.}
The shared predicate \texttt{Riemann\_vonMangoldt\_bound} quantifies over $T\ge 2$.
\paragraph{Source.}
Hasanalizade--Shen--Wong, arXiv:2107.06506, Corollary~1.4, states the
$|S(T)|$ bounds for $T\ge e$.
\paragraph{Proposed correction.}
State the HSW corollaries with an explicit $T\ge e$ quantifier.

\section*{Status in the Lean formalization}
The corresponding statements in
\texttt{PrimeNumberTheoremAnd/IEANTN/TMEEMT.lean} and
\texttt{SecondarySummary.lean} have been aligned with the source papers
for items~\ref{err:dn-11181}, \ref{err:pt2021-psi}, \ref{err:jy-1515},
\ref{err:dkl-loglog}, \ref{err:axler-mandl}, \ref{err:cms-orientation},
\ref{err:trudgian-cor2}, \ref{err:dudek-threshold}, and
\ref{err:cully-hugill-threshold}.
Item~\ref{err:dn-11181} also required restoring the factor $2$ in the
$\pi_2$ main term $\frac{2x^3}{27\log^3 x}$, which Art01 already had
correctly but which had been dropped in an earlier transcription into Lean.
Item~\ref{err:axler-mandl} was likewise present as a Lean transcription
error (copied from the wiki); the formal statements now bound Mandl's
$B_n$ as in Theorems~1.6--1.7 of Axler.
Items~\ref{err:cms-orientation}, \ref{err:trudgian-cor2},
\ref{err:dudek-threshold}, and~\ref{err:cully-hugill-threshold} were
present both on Art09 and in Lean; all now follow the papers.
Items~\ref{err:art08-log-plus-two} and~\ref{err:art08-kadiri-r0} are
wiki-only: the project's \texttt{riemannZeta.classicalZeroFree} already
uses $\log|\Im s|$ without the spurious $+2$, matching the papers and
Fiori--Kadiri--Swaminathan.
Items~\ref{err:hiary-063}, \ref{err:hsw-forms}, \ref{err:patel-log},
\ref{err:rosser-threshold}, \ref{err:trudgian-N-5T},
\ref{err:lehman-threshold}, and~\ref{err:delange-real} are aligned in
Lean in a companion pull request.
Item~\ref{err:ramare-thresholds}: Lean follows the 2023 corrigendum
thresholds (including the $x\ge 115$ line omitted on the wiki).
Items~\ref{err:gourdon-conditional}, \ref{err:schoenfeld-open},
\ref{err:kadiri-lumley-open}, \ref{err:cheng-graham-kusmin}, and
\ref{err:hsw-threshold} are aligned in the present update.
