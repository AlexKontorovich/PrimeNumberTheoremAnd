\begin{abstract}
This note collects discrepancies found while matching statements on the
\href{https://archimede.pages.math.cnrs.fr/tme-emt-wiki/index.html}{TME--EMT wiki}
against the cited source papers, in the course of formalizing those
statements in Lean.  Items are grouped by wiki page.  Only high-confidence
mismatches are listed; each entry gives the wiki text, the source, and a
proposed correction.
\end{abstract}

\section*{How to read this note}
Citations of the form ``Art01'' refer to
\href{https://archimede.pages.math.cnrs.fr/tme-emt-wiki/Art01.html}{Art01.html}
(``Explicit bounds on primes'').  Page numbers for papers are from the
versions available at the URLs recorded in the wiki bibliography.

\section{Art01 --- Explicit bounds on primes}

\subsection{Del\'eglise--Nicolas 2019, $\pi_2$ upper error constant}
\label{err:dn-11181}
\paragraph{Wiki.}
Art01 quotes, for $x\ge 60173$,
\[
\pi_2(x)-\Bigl(\tfrac{x^3}{3\log x}+\tfrac{x^3}{9\log^2 x}+\tfrac{2x^3}{27\log^3 x}\Bigr)
\le \tfrac{11181\,x^3}{648\log^4 x}.
\]
\paragraph{Source.}
Del\'eglise--Nicolas, \emph{The Landau function and the Riemann hypothesis},
arXiv:1907.07664, Corollary~2.7, display~(2.21), uses the constant
\textbf{1181} (not 11181).  The same constant appears in the simplification
leading to~(2.22), as the coefficient $1181/216$.
\paragraph{Proposed correction.}
Replace $11181$ by $1181$.

\subsection{Platt--Trudgian 2021, $\psi$ versus $\pi$}
\label{err:pt2021-psi}
\paragraph{Wiki.}
Art01 attributes to Platt--Trudgian~(2021) the bound
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 235\,(\log x)^{0.52}\exp\Bigl(-\sqrt{\tfrac{\log x}{5.573412}}\Bigr)
\qquad(x\ge e^{2000}).
\]
\paragraph{Source.}
Platt--Trudgian, \emph{The Riemann hypothesis is true up to}~$3\cdot 10^{12}$,
Table~1 at $X=2000$ and Theorem~1 give a $\psi$-bound of the shape
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 411.4\,\Bigl(\tfrac{\log x}{R}\Bigr)^{1.52}
\exp\bigl(-1.89\sqrt{\log x/R}\bigr),\qquad R=5.573412.
\]
The constants $235$ and exponent $0.52$ belong to the companion $\pi$-bound
(Corollary~2), not to the $\psi$-bound.
\paragraph{Proposed correction.}
Quote the $\psi$-bound from Table~1 / Theorem~1, or explicitly relabel the
displayed estimate as a $\pi$-bound and cite Corollary~2.

\subsection{Johnston--Yang 2023, exponent on $\log x$}
\label{err:jy-1515}
\paragraph{Wiki.}
Art01 quotes
\[
\Bigl|\tfrac{\psi(x)-x}{x}\Bigr|
\le 9.39\,(\log x)^{1.51}\exp\bigl(-0.8274\sqrt{\log x}\bigr)
\qquad(x\ge 2).
\]
\paragraph{Source.}
Johnston--Yang, Theorem~1.1, uses the exponent $\mathbf{1.515}$.
\paragraph{Proposed correction.}
Replace $1.51$ by $1.515$.

\subsection{De~Koninck--Letendre 2020, missing $k$ in $\log\log k$}
\label{err:dkl-loglog}
\paragraph{Wiki.}
Art01, \S4, writes
\[
\sum_{i\le k}\log\log p_i
\ge k\Biggl(\log\log k+\frac{\log\log - 3/2}{\log k}\Biggr)
\qquad(k\ge 6),
\]
with a bare $\log\log$ in the numerator.
\paragraph{Source.}
De~Koninck--Letendre, arXiv:1812.09950, Lemma~4.8~(4.21):
\[
\sum_{i\le k}\log\log p_i
\ge k\Biggl(\log\log k+\frac{\log\log k - 3/2}{\log k}\Biggr)
\qquad(k\ge 6).
\]
\paragraph{Proposed correction.}
Replace the bare $\log\log$ by $\log\log k$.

\subsection{B\"uthe 2016, omitted $\pi^*$ bound}
\label{err:buthe-pistar}
\paragraph{Wiki.}
Art01, \S1, under B\"uthe~(2016), lists RH-conditional bounds for
$\psi$, $\vartheta$, and $\pi$ only.
\paragraph{Source.}
B\"uthe also states
\[
\bigl|\pi^*(x)-\mathrm{li}(x)\bigr|
\le \tfrac{\sqrt{x}}{8\pi}\log x
\qquad(x>59),
\]
under the same height hypothesis.
\paragraph{Proposed correction.}
Add the $\pi^*$ bound to the displayed list.

\subsection{Axler 2019, Mandl $B_n$ mislabelled as $\sum p_i$}
\label{err:axler-mandl}
\paragraph{Wiki.}
Art01, \S4, attributes to Axler~(2019) the bounds
\[
\sum_{i\le k}p_i
\;\gtrless\;
\frac{k^2}{4}+\frac{k^2}{4\log k}
-\frac{k^2(\log\log k-c)}{4(\log k)^2}
\]
(with $c=2.9$ for the lower bound at $k\ge 6309751$, and $c=4.42$ for the
upper bound at $k\ge 256376$).
\paragraph{Source.}
Axler, \emph{On the sum of the first $n$ prime numbers},
arXiv:1606.06874, Theorems~1.6 and~1.7 bound Mandl's quantity
\[
B_n=\frac{n\,p_n}{2}-\sum_{k\le n}p_k,
\]
not $\sum p_k$.  The actual prime-sum estimates are Theorems~1.8 and~1.9
(leading factor $n^2/(2\log n)$, different constants and ranges).
\paragraph{Proposed correction.}
Replace $\sum_{i\le k}p_i$ by $B_k$ (and cite Theorems~1.6--1.7), or else
quote Theorems~1.8--1.9 for the prime-sum bounds.

\section{Art06 --- Bounds on $|\zeta(s)|$}
No high-confidence statement mismatches with the cited papers were found
among the $\zeta$-only results recorded in Art06 (Lehman, Cheng--Graham,
Hiary, Backlund, Trudgian, Patel, Ford, Rosser, Hasanalizade--Shen--Wong,
Ramaré, Delange).  Year labels on some theorem heads (``2001'', ``2012'')
are citation noise only and do not affect the displayed inequalities.

\section*{Status in the Lean formalization}
The corresponding statements in
\texttt{PrimeNumberTheoremAnd/IEANTN/TMEEMT.lean} have been aligned with the
source papers for items~\ref{err:dn-11181}, \ref{err:pt2021-psi},
\ref{err:jy-1515}, \ref{err:dkl-loglog}, and~\ref{err:axler-mandl}.
Item~\ref{err:dn-11181} also required restoring the factor $2$ in the
$\pi_2$ main term $\frac{2x^3}{27\log^3 x}$, which Art01 already had
correctly but which had been dropped in an earlier transcription into Lean.
Item~\ref{err:axler-mandl} was likewise present as a Lean transcription
error (copied from the wiki); the formal statements now bound Mandl's
$B_n$ as in Theorems~1.6--1.7 of Axler.
